-- ==========================================
-- Tamil Nadu Election 2026 Analysis
-- SQL Reference Queries
-- ==========================================

-- View 2021 results
SELECT *
FROM tn_2021_results;


-- View 2026 results
SELECT *
FROM tn_2026_results;


-- Constituencies where winner changed
SELECT
    a.ac_number,
    a.constituency,
    a.party AS winner_2021,
    b.party AS winner_2026
FROM tn_2021_results a
JOIN tn_2026_results b
ON a.ac_number = b.ac_number
WHERE a.party <> b.party;


-- Total flipped seats
SELECT COUNT(*) AS flipped_seats
FROM (
    SELECT
        a.ac_number
    FROM tn_2021_results a
    JOIN tn_2026_results b
    ON a.ac_number=b.ac_number
    WHERE a.party<>b.party
) x;


-- Regional flip intensity
SELECT
region,
COUNT(*) AS total_constituencies,
SUM(flipped) AS flipped_seats,
ROUND(
SUM(flipped)*100.0/
COUNT(*),
2
) AS flip_rate
FROM constituency_flips
GROUP BY region
ORDER BY flip_rate DESC;


-- Party seat comparison
SELECT
party,
COUNT(*) AS seats_2026
FROM tn_2026_results
GROUP BY party
ORDER BY seats_2026 DESC;


-- Closest constituencies
SELECT
constituency,
winner_party,
runner_up_party,
victory_margin
FROM margins_2026
ORDER BY victory_margin ASC
LIMIT 10;


-- Regional dominance
SELECT
region,
party,
COUNT(*) AS seats
FROM constituency_flips
GROUP BY region,party
ORDER BY region;
